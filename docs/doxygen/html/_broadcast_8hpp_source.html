42 static void broadcastExistingPlayers(

44 int newPlayerID) {
45 auto players = game.getAllPlayers();
46
47 for (const auto &player : players) {
48 if (player && player->isConnected() &&
49 player->getPlayerId() != newPlayerID) {
50 std::pair<float, float> pos = player->getPosition();
51 float speed = player->getSpeed();
52 int health = player->getHealth().value_or(0);
53
54 auto existPlayerPacket = PacketBuilder::makeNewPlayer(
55 player->getPlayerId(), pos.first, pos.second, speed, health);
56 networkManager.sendToClient(
